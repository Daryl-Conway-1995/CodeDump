 //create array of people
        Person sam = new Person("sam", 23, "cleaner");
        Person lucy = new Person("lucy", 28, "doctor");
        Person thor = new Person("thor", 485, "superhero");
        Person nick = new Person("nick", 34 , "biologist");

        ArrayList<String> results = new ArrayList<>();
        ArrayList<Person> people= new ArrayList<>();
        people.add(sam);
        people.add(lucy);
        people.add(thor);
        people.add(nick);

        File file = new File("read and write.txt");
        int size =0;
        char[] message = new char[130];
        String concatenateMessage = "";

        //create file reader and writer
        try
        {
            FileWriter fw = new FileWriter(file);
            FileReader fr = new FileReader(file);
            for (Person p:people)
            {
                //add data to text file
                fw.write(p.getInfo()+"\n");
                fw.flush();
            }
            fw.close();
            size = fr.read(message);
            System.out.println(size);
            for (char c:message)
            {
                //read data from text file
                concatenateMessage = concatenateMessage + c;
            }
            fr.close();

        }
        catch
                (Exception e)
        {e.printStackTrace();}

        //separate string to different people
        for (String segment:concatenateMessage.split("\n"))
        {
            results.add(segment);
        }

        //print results
        int count =0;
        for (String print:results)
        {
            System.out.println(count + ": "+print);
            count++;
        }
